class NavigationView extends Backbone.View
  id: 'NavigationView'

  events:
    'click li': 'navigate'

  render: ->
    $(@el).html(bt.templates.navigation())
    return @

  navigate: (e) ->
    @$('li').removeClass('selected')
    $(e.currentTarget).addClass('selected')
